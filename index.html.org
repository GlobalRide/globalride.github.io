			function initMap() {

				var latlng_osaka = new google.maps.LatLng(34.686856, 135.499253);//中心の緯度, 経度

				});
				var map_osaka = new google.maps.Map(document.getElementById('map_osaka'), {
					zoom: 14,//ズームの調整
					center: latlng_osaka//上で設定した中心
				});

				//マーカーの設定
				var marker_tokyo = new google.maps.Marker({
					position: latlng_tokyo,
					map: map_tokyo,
					icon: 'icon_pin_tokyo.png'//ここを追加
				});
				var marker_osaka = new google.maps.Marker({
